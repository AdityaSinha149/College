1) Grammar (Q1)

- Grammar: 
    - Program → main ( ) { declarations statement-list }
    - declarations → data-type identifier-list ; declarations | ε
    - data-type → int | char
    - identifier-list → id | id , identifier-list | id [ num ] , identifier-list | id [ num ]
    - statement-list → statement ; statement-list | ε
    - statement → assign-stat | decision-stat | looping-stat
    - assign-stat → id = expn
    - expn → simple-expn eprime
    - eprime → relop simple-expn | ε
    - simple-expn → term seprime
    - seprime → addop term seprime | ε
    - term → factor tprime
    - tprime → mulop factor tprime | ε
    - factor → id | num
    - decision-stat → if ( expn ) { statement-list } dprime
    - dprime → else { statement-list } | ε
    - looping-stat → while ( expn ) { statement-list }
    - relop → == | != | <= | >= | > | <
    - addop → + | -
    - mulop → * | / | %

static void initTokenStream(TokenStream *stream) {
    stream->head = NULL;
    stream->tail = NULL;
}

static void freeTokenStream(TokenStream *stream) {
    TokenNode *currentNode = stream->head;

    while (currentNode != NULL) {
        TokenNode *nextNode = currentNode->next;

        free(currentNode);
        currentNode = nextNode;
    }

    stream->head = NULL;
    stream->tail = NULL;
}

static int appendToken(TokenStream *stream, token currentToken) {
    TokenNode *newNode = (TokenNode *)malloc(sizeof(TokenNode));

    if (newNode == NULL)
        return 0;

    newNode->value = currentToken;
    newNode->next = NULL;

    if (stream->tail == NULL) {
        stream->head = newNode;
        stream->tail = newNode;
    } else {
        stream->tail->next = newNode;
        stream->tail = newNode;
    }

    return 1;
}

static int tokenizeSourceFile(const char *fileName, TokenStream *stream, char *error, size_t errorSize) {
    FILE *src = fopen(fileName, "r");
    FILE *tmp;
    int row = 1;
    int col = 1;

    if (src == NULL) {
        snprintf(error, errorSize, "Unable to open input file '%s'.", fileName);
        return 0;
    }

    tmp = tmpfile();
    if (tmp == NULL) {
        fclose(src);
        snprintf(error, errorSize, "Unable to create a temporary file for preprocessing.");
        return 0;
    }

    macroCount = 0;
    type[0] = '\0';

    preprocess(src, tmp);
    rewind(tmp);

    while (1) {
        token currentToken = getNextToken(tmp, &row, &col);

        if (currentToken.tokenName[0] == '\0')
            break;

        if (!appendToken(stream, currentToken)) {
            fclose(src);
            fclose(tmp);
            snprintf(error, errorSize, "Out of memory while storing tokens.");
            return 0;
        }
    }

    fclose(src);
    fclose(tmp);
    return 1;
}

static const token *peekToken(const Parser *parser) {
    (void)parser->stream;

    if (parser->currentNode == NULL)
        return NULL;

    return &parser->currentNode->value;
}

static const token *advanceToken(Parser *parser) {
    const token *currentToken = peekToken(parser);

    if (currentToken != NULL)
        parser->currentNode = parser->currentNode->next;

    return currentToken;
}

static int isAtEnd(const Parser *parser) {
    return parser->currentNode == NULL;
}

static void setParserError(Parser *parser, const char *message) {
    const token *currentToken;

    if (parser->error[0] != '\0')
        return;

    currentToken = peekToken(parser);
    if (currentToken == NULL) {
        snprintf(parser->error, sizeof(parser->error), "Parse error at end of file: %s", message);
        return;
    }

    snprintf(parser->error,
             sizeof(parser->error),
             "Parse error at line %d, column %d near '%s': %s",
             currentToken->row,
             currentToken->col,
             currentToken->tokenValue[0] ? currentToken->tokenValue : currentToken->tokenName,
             message);
}

static int checkKeyword(const Parser *parser, const char *keyword) {
    const token *currentToken = peekToken(parser);

    return currentToken != NULL &&
           strcmp(currentToken->tokenName, "keyword") == 0 &&
           strcmp(currentToken->tokenValue, keyword) == 0;
}

static int checkIdentifier(const Parser *parser) {
    const token *currentToken = peekToken(parser);

    return currentToken != NULL && strcmp(currentToken->tokenName, "id") == 0;
}

static int checkNumber(const Parser *parser) {
    const token *currentToken = peekToken(parser);

    return currentToken != NULL && strcmp(currentToken->tokenName, "num") == 0;
}

static int checkSymbol(const Parser *parser, const char *symbol) {
    const token *currentToken = peekToken(parser);

    return currentToken != NULL &&
           strcmp(currentToken->tokenName, "symbol") == 0 &&
           strcmp(currentToken->tokenValue, symbol) == 0;
}

static int checkAssignmentOp(const Parser *parser, const char *value) {
    const token *currentToken = peekToken(parser);

    return currentToken != NULL &&
           strcmp(currentToken->tokenName, "assignOp") == 0 &&
           strcmp(currentToken->tokenValue, value) == 0;
}

static int checkRelOp(const Parser *parser) {
    const token *currentToken = peekToken(parser);

    if (currentToken == NULL || strcmp(currentToken->tokenName, "relOp") != 0)
        return 0;

    return strcmp(currentToken->tokenValue, "==") == 0 ||
           strcmp(currentToken->tokenValue, "!=") == 0 ||
           strcmp(currentToken->tokenValue, "<=") == 0 ||
           strcmp(currentToken->tokenValue, ">=") == 0 ||
           strcmp(currentToken->tokenValue, "<") == 0 ||
           strcmp(currentToken->tokenValue, ">") == 0;
}

static int checkAddOp(const Parser *parser) {
    const token *currentToken = peekToken(parser);

    if (currentToken == NULL || strcmp(currentToken->tokenName, "ariOp") != 0)
        return 0;

    return strcmp(currentToken->tokenValue, "+") == 0 ||
           strcmp(currentToken->tokenValue, "-") == 0;
}

static int checkMulOp(const Parser *parser) {
    const token *currentToken = peekToken(parser);

    if (currentToken == NULL || strcmp(currentToken->tokenName, "ariOp") != 0)
        return 0;

    return strcmp(currentToken->tokenValue, "*") == 0 ||
           strcmp(currentToken->tokenValue, "/") == 0 ||
           strcmp(currentToken->tokenValue, "%") == 0;
}

static int consumeKeyword(Parser *parser, const char *keyword, const char *message) {
    if (!checkKeyword(parser, keyword)) {
        setParserError(parser, message);
        return 0;
    }

    advanceToken(parser);
    return 1;
}

static int consumeIdentifier(Parser *parser, const char *message, token *captured) {
    const token *currentToken;

    if (!checkIdentifier(parser)) {
        setParserError(parser, message);
        return 0;
    }

    currentToken = advanceToken(parser);
    if (captured != NULL)
        *captured = *currentToken;

    return 1;
}

static int consumeNumber(Parser *parser, const char *message, token *captured) {
    const token *currentToken;

    if (!checkNumber(parser)) {
        setParserError(parser, message);
        return 0;
    }

    currentToken = advanceToken(parser);
    if (captured != NULL)
        *captured = *currentToken;

    return 1;
}

static int consumeAssignmentOp(Parser *parser, const char *value, const char *message) {
    if (!checkAssignmentOp(parser, value)) {
        setParserError(parser, message);
        return 0;
    }

    advanceToken(parser);
    return 1;
}

static const char *parseProgramReturnType(Parser *parser) {
    if (checkKeyword(parser, "int")) {
        advanceToken(parser);
        return "int";
    }

    if (checkKeyword(parser, "void")) {
        advanceToken(parser);
        return "void";
    }

    return "void";
}

static void addDeclaredIdentifier(Parser *parser, const char *typeName, const token *identifier, int arrayLength) {
    int size = findSizeOf((char *)typeName);

    if (arrayLength > 0)
        size *= arrayLength;

    addVariableSymbol(parser->localTable, identifier->tokenValue, typeName, size);
}

static int parseDeclaration(Parser *parser) {
    const char *typeName;
    token identifier;

    typeName = peekToken(parser)->tokenValue;
    advanceToken(parser);

    if (!consumeIdentifier(parser, "Expected identifier in declaration.", &identifier))
        return 0;

    if (checkSymbol(parser, "[")) {
        token arrayLengthToken;

        advanceToken(parser);
        if (!consumeNumber(parser, "Expected array size after '['.", &arrayLengthToken))
            return 0;

        if (!checkSymbol(parser, "]")) {
            setParserError(parser, "Expected ']' after array size.");
            return 0;
        }
        advanceToken(parser);

        addDeclaredIdentifier(parser, typeName, &identifier, atoi(arrayLengthToken.tokenValue));
    } else {
        addDeclaredIdentifier(parser, typeName, &identifier, 0);
    }

    while (checkSymbol(parser, ",")) {
        advanceToken(parser);

        if (!consumeIdentifier(parser, "Expected identifier after ','.", &identifier))
            return 0;

        if (checkSymbol(parser, "[")) {
            token arrayLengthToken;

            advanceToken(parser);
            if (!consumeNumber(parser, "Expected array size after '['.", &arrayLengthToken))
                return 0;

            if (!checkSymbol(parser, "]")) {
                setParserError(parser, "Expected ']' after array size.");
                return 0;
            }
            advanceToken(parser);

            addDeclaredIdentifier(parser, typeName, &identifier, atoi(arrayLengthToken.tokenValue));
        } else {
            addDeclaredIdentifier(parser, typeName, &identifier, 0);
        }
    }

    if (!checkSymbol(parser, ";")) {
        setParserError(parser, "Expected ';' after declaration.");
        return 0;
    }

    advanceToken(parser);
    return 1;
}

static int parseDeclarations(Parser *parser) {
    while (checkKeyword(parser, "int") || checkKeyword(parser, "char"))
        if (!parseDeclaration(parser))
            return 0;

    return 1;
}

static int parseFactor(Parser *parser) {
    if (checkIdentifier(parser)) {
        advanceToken(parser);
        return 1;
    }

    if (checkNumber(parser)) {
        advanceToken(parser);
        return 1;
    }

    setParserError(parser, "Expected identifier or number in expression.");
    return 0;
}

static int parseTerm(Parser *parser) {
    if (!parseFactor(parser))
        return 0;

    while (checkMulOp(parser)) {
        advanceToken(parser);
        if (!parseFactor(parser))
            return 0;
    }

    return 1;
}

static int parseSimpleExpn(Parser *parser) {
    if (!parseTerm(parser))
        return 0;

    while (checkAddOp(parser)) {
        advanceToken(parser);
        if (!parseTerm(parser))
            return 0;
    }

    return 1;
}

static int parseExpn(Parser *parser) {
    if (!parseSimpleExpn(parser))
        return 0;

    if (checkRelOp(parser)) {
        advanceToken(parser);
        if (!parseSimpleExpn(parser))
            return 0;
    }

    return 1;
}

static int parseAssignStat(Parser *parser) {
    if (!consumeIdentifier(parser, "Expected identifier in assignment statement.", NULL))
        return 0;

    if (!consumeAssignmentOp(parser, "=", "Expected '=' in assignment statement."))
        return 0;

    return parseExpn(parser);
}

static int parseDecisionStat(Parser *parser) {
    if (!consumeKeyword(parser, "if", "Expected 'if' keyword."))
        return 0;

    if (!checkSymbol(parser, "(")) {
        setParserError(parser, "Expected '(' after 'if'.");
        return 0;
    }
    advanceToken(parser);

    if (!parseExpn(parser))
        return 0;

    if (!checkSymbol(parser, ")")) {
        setParserError(parser, "Expected ')' after condition.");
        return 0;
    }
    advanceToken(parser);

    if (!checkSymbol(parser, "{")) {
        setParserError(parser, "Expected '{' to start if-block.");
        return 0;
    }
    advanceToken(parser);

    if (!parseStatementList(parser))
        return 0;

    if (!checkSymbol(parser, "}")) {
        setParserError(parser, "Expected '}' to close if-block.");
        return 0;
    }
    advanceToken(parser);

    if (checkKeyword(parser, "else")) {
        advanceToken(parser);

        if (!checkSymbol(parser, "{")) {
            setParserError(parser, "Expected '{' to start else-block.");
            return 0;
        }
        advanceToken(parser);

        if (!parseStatementList(parser))
            return 0;

        if (!checkSymbol(parser, "}")) {
            setParserError(parser, "Expected '}' to close else-block.");
            return 0;
        }
        advanceToken(parser);
    }

    return 1;
}

static int parseLoopingStat(Parser *parser) {
    if (!consumeKeyword(parser, "while", "Expected 'while' keyword."))
        return 0;

    if (!checkSymbol(parser, "(")) {
        setParserError(parser, "Expected '(' after 'while'.");
        return 0;
    }
    advanceToken(parser);

    if (!parseExpn(parser))
        return 0;

    if (!checkSymbol(parser, ")")) {
        setParserError(parser, "Expected ')' after loop condition.");
        return 0;
    }
    advanceToken(parser);

    if (!checkSymbol(parser, "{")) {
        setParserError(parser, "Expected '{' to start loop block.");
        return 0;
    }
    advanceToken(parser);

    if (!parseStatementList(parser))
        return 0;

    if (!checkSymbol(parser, "}")) {
        setParserError(parser, "Expected '}' to close loop block.");
        return 0;
    }
    advanceToken(parser);

    return 1;
}

static int parseStatement(Parser *parser) {
    if (checkIdentifier(parser)) {
        if (!parseAssignStat(parser))
            return 0;

        if (!checkSymbol(parser, ";")) {
            setParserError(parser, "Expected ';' after assignment statement.");
            return 0;
        }

        advanceToken(parser);
        return 1;
    }

    if (checkKeyword(parser, "if"))
        return parseDecisionStat(parser);

    if (checkKeyword(parser, "while"))
        return parseLoopingStat(parser);

    setParserError(parser, "Expected assignment, if-statement, or while-statement.");
    return 0;
}

static int parseStatementList(Parser *parser) {
    while (!isAtEnd(parser) && !checkSymbol(parser, "}"))
        if (!parseStatement(parser))
            return 0;

    return 1;
}

static int parseProgram(Parser *parser) {
    const char *returnType;
    token mainIdentifier;
    symbolTableEntry *mainEntry;

    returnType = parseProgramReturnType(parser);

    if (!consumeIdentifier(parser, "Expected 'main' at the start of the program.", &mainIdentifier))
        return 0;

    if (strcmp(mainIdentifier.tokenValue, "main") != 0) {
        setParserError(parser, "Program must start with 'main'.");
        return 0;
    }

    mainEntry = addFunctionSymbol(parser->globalTable, "main", returnType);
    if (mainEntry == NULL || mainEntry->local == NULL) {
        setParserError(parser, "Unable to create symbol table entry for main.");
        return 0;
    }

    parser->localTable = mainEntry->local;

    if (!checkSymbol(parser, "(")) {
        setParserError(parser, "Expected '(' after main.");
        return 0;
    }
    advanceToken(parser);

    if (!checkSymbol(parser, ")")) {
        setParserError(parser, "Expected ')' after '('.");
        return 0;
    }
    advanceToken(parser);

    if (!checkSymbol(parser, "{")) {
        setParserError(parser, "Expected '{' to start program body.");
        return 0;
    }
    advanceToken(parser);

    if (!parseDeclarations(parser))
        return 0;
    if (!parseStatementList(parser))
        return 0;

    if (!checkSymbol(parser, "}")) {
        setParserError(parser, "Expected '}' to close program body.");
        return 0;
    }
    advanceToken(parser);

    if (!isAtEnd(parser)) {
        setParserError(parser, "Unexpected tokens after the end of the program.");
        return 0;
    }

    return 1;
}

int main(int argc, char *argv[]) {
    TokenStream stream;
    Parser parser;
    symbolTable globalTable;
    char fileName[256];
    char error[256] = "";
    int parseStatus;

    initTokenStream(&stream);
    initSymbolTable(&globalTable);

    if (argc >= 2) {
        strncpy(fileName, argv[1], sizeof(fileName) - 1);
        fileName[sizeof(fileName) - 1] = '\0';
    } else {
        printf("Enter input C file: ");
        if (scanf("%255s", fileName) != 1) {
            fprintf(stderr, "Unable to read the input file name.\n");
            return 1;
        }
    }

    if (!tokenizeSourceFile(fileName, &stream, error, sizeof(error))) {
        fprintf(stderr, "%s\n", error);
        freeTokenStream(&stream);
        freeSymbolTable(&globalTable);
        return 1;
    }

    memset(&parser, 0, sizeof(parser));
    parser.stream = &stream;
    parser.currentNode = stream.head;
    parser.globalTable = &globalTable;

    parseStatus = parseProgram(&parser);

    if (!parseStatus) {
        fprintf(stderr, "%s\n", parser.error);
        freeTokenStream(&stream);
        freeSymbolTable(&globalTable);
        return 1;
    }

    printf("Parsing result: Input program matches the grammar.\n");

    freeTokenStream(&stream);
    freeSymbolTable(&globalTable);
    return 0;
}
